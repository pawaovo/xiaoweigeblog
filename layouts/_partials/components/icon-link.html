{{/* 图标链接组件 参数： - .url: 链接URL (如果为空则显示禁用状态) - .icon: 图标名称 (demo, github, calendar) - .title: 工具提示文本 - .event: 统计事件名称 - .text: 显示的文本 (可选，用于日期等) */}} {{ $hasUrl := .url }} {{ $iconName := .icon }} {{ $title := .title }} {{ $event := .event }} {{ $text := .text }} {{ if $hasUrl }} {{ partial "components/svg-icons" (dict "icon" $iconName) }} {{ if $text }}{{ $text }}{{ end }} {{ else }} {{ partial "components/svg-icons" (dict "icon" $iconName) }} {{ if $text }}{{ $text }}{{ end }} {{ end }}
